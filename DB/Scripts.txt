CREATE TABLE ECOM_BACKOFFICE_USERS(
	BACKOFFICE_USER_ID int AUTO_INCREMENT PRIMARY KEY,
    USERNAME varchar(256) ,
    PASSWORD varchar(512) ,
    FIRST_NAME varchar(256) ,
    LAST_NAME varchar(256),
    EMP_ID VARCHAR(128) ,
    LAST_LOGIN_DATE datetime ,
    IS_ENABLED varchar(2) ,
    IS_DELETED varchar(2) ,
    IS_LOGGED_IN varchar(2) ,
    LAST_LOGIN_IP varchar(64) ,
    CREATED_DATE datetime ,
    UPDATED_DATE datetime 
    );
CREATE TABLE ECOM_USER_CONTACTS (
    USER_CONTACT_ID INT AUTO_INCREMENT PRIMARY KEY,
    USER_ID INT,
    EMAIL VARCHAR(512) NULL,
    MOBILE_NO VARCHAR(128) NULL,
    R_ADDRESS VARCHAR(256) NULL,
    IS_MOBILE_VERIFIED VARCHAR(2) NULL,
    IS_EMAIL_VERIFIED VARCHAR(2) NULL,
    CONSTRAINT FK_USER_ID FOREIGN KEY (USER_ID) REFERENCES ECOM.ecom_backoffice_users(BACKOFFICE_USER_ID)
);


create table ECOM_PRODUCT_CATEGORIES(
	PRODUCT_CATEGORY_ID int AUTO_INCREMENT PRIMARY KEY,
    PRODUCT_CATEGORY_NAME VARCHAR(256),
    PRODUCT_CATEGORY_TYPE VARCHAR(256),
	IS_ACTIVE VARCHAR(2),
    IS_DELETED VARCHAR(2),
    INSERTED_DATE datetime,
    UPDATED_DATE DATETIME,
    INSERTED_BY int ,
    UPDATED_BY INT ,
    CONSTRAINT FK_INSERTED_BY FOREIGN KEY (INSERTED_BY) REFERENCES ECOM.ecom_backoffice_users(BACKOFFICE_USER_ID),
	CONSTRAINT FK_UPDATED_BY FOREIGN KEY (UPDATED_BY) REFERENCES ECOM.ecom_backoffice_users(BACKOFFICE_USER_ID)

);
